## 📌 5. Frontend API 정의 (Axios)
📁 api/index.js
```
import axios from 'axios'

const api = axios.create({
  baseURL: '/api',
  headers: { 'Content-Type': 'application/json' }
})
```

🏨 Room API
```
export const roomApi = {

  // 전체 객실 조회 (날짜 조건 optional)
  getAll: (checkIn, checkOut) =>
    api.get('/rooms', {
      params: { checkIn, checkOut }
    }),

  // 객실 상세 조회
  getById: (id) =>
    api.get(`/rooms/${id}`)
}
```

🧾 Reservation API
```
export const reservationApi = {

  // 예약 생성
  create: (data) =>
    api.post('/reservations', data),

  // 이메일로 예약 조회
  getMyReservations: (email) =>
    api.get('/reservations/my', {
      params: { email }
    }),

  // 예약 취소
  cancel: (id) =>
    api.patch(`/reservations/${id}/cancel`)
}
```
